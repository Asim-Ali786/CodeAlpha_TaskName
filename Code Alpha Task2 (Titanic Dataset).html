import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import PercentFormatter

sns.set_style("whitegrid")
plt.rcParams['figure.figsize'] = (10, 6)
plt.rcParams['font.size'] = 12

df = pd.read_csv('Titanic-Dataset.csv')

df['FamilySize'] = df['SibSp'] + df['Parch'] + 1
df['Title'] = df['Name'].str.extract(' ([A-Za-z]+)\.', expand=False)
df['AgeGroup'] = pd.cut(df['Age'], bins=[0, 12, 18, 30, 50, 100], 
                        labels=['Child', 'Teen', 'Young Adult', 'Adult', 'Senior'])
plt.figure(figsize=(8, 5))
sns.barplot(x='Pclass', y='Survived', data=df, errorbar=None, palette='viridis')
plt.title('Survival Rate by Passenger Class', pad=20)
plt.xlabel('Passenger Class')
plt.ylabel('Survival Rate')
plt.gca().yaxis.set_major_formatter(PercentFormatter(1.0))
plt.show()
plt.figure(figsize=(10, 6))
sns.histplot(data=df, x='Age', hue='Survived', bins=30, kde=True, 
             palette={0: 'red', 1: 'green'}, alpha=0.6)
plt.title('Age Distribution by Survival Status', pad=20)
plt.xlabel('Age')
plt.ylabel('Count')
plt.legend(title='Survived', labels=['No', 'Yes'])
plt.show()
plt.figure(figsize=(10, 6))
sns.boxplot(x='Embarked', y='Fare', data=df, palette='Set2')
plt.title('Fare Distribution by Embarkation Port', pad=20)
plt.xlabel('Embarkation Port')
plt.ylabel('Fare')
plt.yscale('log')  # Using log scale due to outliers
plt.show()
plt.figure(figsize=(12, 6))
sns.barplot(x='AgeGroup', y='Survived', hue='Sex', data=df, 
            errorbar=None, palette='coolwarm')
plt.title('Survival Rate by Age Group and Gender', pad=20)
plt.xlabel('Age Group')
plt.ylabel('Survival Rate')
plt.gca().yaxis.set_major_formatter(PercentFormatter(1.0))
plt.legend(title='Gender')
plt.show()
plt.figure(figsize=(10, 8))
corr = df.corr(numeric_only=True)
mask = np.triu(np.ones_like(corr, dtype=bool))
sns.heatmap(corr, mask=mask, annot=True, cmap='coolwarm', 
            center=0, linewidths=.5)
plt.title('Feature Correlation Heatmap', pad=20)
plt.show()
fig, axes = plt.subplots(2, 2, figsize=(15, 12))
fig.suptitle('Titanic Dataset Key Insights', y=1.02, fontsize=16)

# Panel 1: Survival by Class
sns.barplot(ax=axes[0, 0], x='Pclass', y='Survived', data=df, 
            errorbar=None, palette='viridis')
axes[0, 0].set_title('Survival Rate by Passenger Class')
axes[0, 0].yaxis.set_major_formatter(PercentFormatter(1.0))

# Panel 2: Age Distribution
sns.histplot(ax=axes[0, 1], data=df, x='Age', hue='Survived', 
             bins=30, kde=True, palette={0: 'red', 1: 'green'}, alpha=0.6)
axes[0, 1].set_title('Age Distribution by Survival Status')
axes[0, 1].legend(title='Survived', labels=['No', 'Yes'])

# Panel 3: Family Size Impact
sns.barplot(ax=axes[1, 0], x='FamilySize', y='Survived', data=df, 
            errorbar=None, palette='rocket')
axes[1, 0].set_title('Survival Rate by Family Size')
axes[1, 0].yaxis.set_major_formatter(PercentFormatter(1.0))

# Panel 4: Fare vs Age
sns.scatterplot(ax=axes[1, 1], data=df, x='Age', y='Fare', 
                hue='Survived', palette={0: 'red', 1: 'green'}, alpha=0.6)
axes[1, 1].set_title('Fare vs Age by Survival Status')
axes[1, 1].legend(title='Survived', labels=['No', 'Yes'])

plt.tight_layout()
plt.show()
# Prepare summary data
summary = df.groupby(['Pclass', 'Sex'])['Survived'].agg(['mean', 'count'])
summary = summary.reset_index()
summary.columns = ['Pclass', 'Sex', 'SurvivalRate', 'PassengerCount']

plt.figure(figsize=(12, 8))
sns.scatterplot(data=summary, x='Pclass', y='SurvivalRate', 
                size='PassengerCount', hue='Sex', 
                sizes=(100, 1000), alpha=0.8, palette='deep')

plt.title('Titanic Survival Story: Class and Gender Matter Most', pad=20)
plt.xlabel('Passenger Class (1 = Highest)')
plt.ylabel('Survival Rate')
plt.yticks(np.arange(0, 1.1, 0.1))
plt.gca().yaxis.set_major_formatter(PercentFormatter(1.0))
plt.grid(True, linestyle='--', alpha=0.3)

# Add annotations
for i, row in summary.iterrows():
    plt.text(row['Pclass']+0.05, row['SurvivalRate'], 
             f"{row['SurvivalRate']:.0%}", 
             ha='left', va='center')

plt.legend(title='Gender', bbox_to_anchor=(1.05, 1), loc='upper left')
plt.show()
